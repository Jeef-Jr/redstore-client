function vRP.ClearInventory(user_id)
	local source = vRP.userSource(user_id)
	local Datatable = vRP.getDatatable(user_id)
	if source and Datatable["inventory"] then
		dataTable["inventory"] = {}
		TriggerEvent("inventory:clearWeapons", user_id)
	end
end

function vRP.Hierarchy(perm)
	local tableList = {}
	for k, v in pairs(permissions[perm]) do
		table.insert(tableList, v)
	end
	return tableList
end
